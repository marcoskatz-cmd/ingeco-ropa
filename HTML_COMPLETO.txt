  <script>
(function () {
  'use strict';

  /* ===================== Estado ===================== */
  var S = {
    token: null,
    usuario: '',
    config: null,
    listas: {},
    seccion: 'corrida',
    exclusiones: {}   // { "LEGAJO|PRENDA": true } marcado como "no entregar"
  };

  var SECCIONES = [
    { id: 'entrega',   label: 'Registrar entrega' },
    { id: 'corrida',   label: 'Corrida actual' },
    { id: 'prevision', label: 'Previsión' },
    { id: 'pedido',    label: 'Armar pedido' },
    { id: 'rrhh',      label: 'Cruce RRHH' },
    { id: 'historial', label: 'Historial' },
    { id: 'roturas',   label: 'Roturas' },
    { id: 'personal',  label: 'Personal' }
  ];

  /* ===================== Puente con el servidor ===================== */
  function rawCall(metodo, payloadObj) {
    return new Promise(function (resolve, reject) {
      if (!(window.google && google.script && google.script.run)) {
        reject(new Error('Esta vista necesita estar desplegada como Web App de Apps Script.'));
        return;
      }
      google.script.run
        .withSuccessHandler(function (jsonStr) {
          var res;
          try { res = JSON.parse(jsonStr); }
          catch (e) { reject(new Error('Respuesta inválida del servidor.')); return; }
          if (res && res.ok) resolve(res.data);
          else reject(new Error(res && res.error ? res.error : 'Error desconocido.'));
        })
        .withFailureHandler(function (err) {
          reject(new Error(err && err.message ? err.message : String(err)));
        })
        .apiCall(metodo, JSON.stringify(payloadObj || {}));
    });
  }
  function api(metodo, payload) {
    payload = payload || {};
    if (S.token) payload.__token = S.token;
    return rawCall(metodo, payload);
  }

  /* ===================== Helpers DOM / formato ===================== */
  function $(id) { return document.getElementById(id); }
  function esc(s) {
    return String(s === null || s === undefined ? '' : s)
      .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;').replace(/'/g, '&#39;');
  }
  function fmt(n) {
    var x = Number(n);
    if (!isFinite(x)) return '0';
    return x.toLocaleString('es-AR');
  }
  function money(n) {
    if (n === null || n === undefined || n === '') return '—';
    var x = Number(n);
    if (!isFinite(x)) return '—';
    return '$ ' + x.toLocaleString('es-AR', { maximumFractionDigits: 2 });
  }
  var toastT;
  function toast(msg, tipo) {
    var t = $('toast');
    t.textContent = msg;
    t.className = 'toast ' + (tipo || '');
    clearTimeout(toastT);
    toastT = setTimeout(function () { t.classList.add('oculto'); }, 3400);
  }
  function tabla(headers, rowsHtml) {
    return '<div class="tabla-wrap"><table class="tabla"><thead><tr>' +
      headers.map(function (h) { return '<th>' + h + '</th>'; }).join('') +
      '</tr></thead><tbody>' + (rowsHtml || '') + '</tbody></table></div>';
  }
  function cargando() { return '<div class="cargando">Cargando…</div>'; }
  function vacio(msg) { return '<div class="vacio">' + esc(msg || 'Sin datos.') + '</div>'; }
  function hoyInput() {
    var d = new Date();
    return d.getFullYear() + '-' + ('0' + (d.getMonth() + 1)).slice(-2) + '-' + ('0' + d.getDate()).slice(-2);
  }
  function inputAFecha(v) {
    if (!v) return '';
    var p = String(v).split('-');
    return p.length === 3 ? (p[2] + '/' + p[1] + '/' + p[0]) : v;
  }
  function badgeEstado(estado, innegociable) {
    var e = String(estado || '');
    if (e === 'CRITICO') return '<span class="badge critico">CRÍTICO</span>' +
      (innegociable ? ' <span class="badge inneg">INNEGOCIABLE</span>' : '');
    if (e === 'ELEGIBLE') return '<span class="badge elegible">ELEGIBLE</span>';
    return '<span class="badge sindato">SIN DATO</span>';
  }

  /* ===================== Modal ===================== */
  function abrirModal(titulo, html) {
    $('modalTitulo').textContent = titulo;
    $('modalCuerpo').innerHTML = html;
    $('modal').classList.remove('oculto');
  }
  function cerrarModal() {
    $('modal').classList.add('oculto');
    $('modalCuerpo').innerHTML = '';
  }

  /* ===================== Login ===================== */
  function initLogin() {
    rawCall('listarNombresLogin', {}).then(function (nombres) {
      var sel = $('loginNombre');
      sel.innerHTML = (nombres || []).map(function (n) {
        return '<option value="' + esc(n) + '">' + esc(n) + '</option>';
      }).join('');
    }).catch(function (e) {
      mostrarLoginError(e.message);
    });
  }
  function mostrarLoginError(msg) {
    var d = $('loginError');
    d.textContent = msg;
    d.classList.remove('oculto');
  }
  function hacerLogin() {
    var nombre = $('loginNombre').value;
    var pin = $('loginPin').value;
    $('loginError').classList.add('oculto');
    if (!nombre || !pin) { mostrarLoginError('Elegí usuario y escribí el PIN.'); return; }
    $('btnLogin').disabled = true;
    rawCall('login', { nombre: nombre, pin: pin }).then(function (res) {
      S.token = res.token;
      S.usuario = res.nombre;
      try { sessionStorage.setItem('ropa_token', res.token); sessionStorage.setItem('ropa_user', res.nombre); } catch (e) {}
      iniciarApp();
    }).catch(function (e) {
      $('btnLogin').disabled = false;
      mostrarLoginError(e.message);
    });
  }
  function salir() {
    S.token = null; S.usuario = '';
    try { sessionStorage.removeItem('ropa_token'); sessionStorage.removeItem('ropa_user'); } catch (e) {}
    $('app').classList.add('oculto');
    $('login').classList.remove('oculto');
    $('btnLogin').disabled = false;
    $('loginPin').value = '';
  }

  /* ===================== Arranque de la app ===================== */
  function iniciarApp() {
    $('login').classList.add('oculto');
    $('app').classList.remove('oculto');
    $('usuarioActual').textContent = S.usuario;
    construirNav();
    Promise.all([api('getConfig'), api('getListas')]).then(function (r) {
      S.config = r[0]; S.listas = r[1] || {};
      ir(S.seccion);
    }).catch(function (e) {
      if (/Sesión|sesión|venció/.test(e.message)) { salir(); return; }
      $('vista').innerHTML = '<div class="error">' + esc(e.message) + '</div>';
    });
  }
  function construirNav() {
    $('nav').innerHTML = SECCIONES.map(function (s) {
      return '<button data-sec="' + s.id + '">' + esc(s.label) + '</button>';
    }).join('');
    Array.prototype.forEach.call($('nav').querySelectorAll('button'), function (b) {
      b.addEventListener('click', function () { ir(b.getAttribute('data-sec')); });
    });
  }
  function marcarNav() {
    Array.prototype.forEach.call($('nav').querySelectorAll('button'), function (b) {
      b.classList.toggle('activo', b.getAttribute('data-sec') === S.seccion);
    });
  }
  function ir(sec) {
    S.seccion = sec;
    marcarNav();
    var fn = RENDER[sec];
    if (fn) fn();
  }
  function manejar(e) { // manejador de errores común para las promesas de sección
    if (/Sesión|sesión|venció/.test(e.message)) { toast('Sesión vencida, volvé a entrar.', 'err'); salir(); return; }
    toast(e.message, 'err');
  }

  /* ===================== Secciones ===================== */
  var RENDER = {};

  /* ---------- Registrar entrega ---------- */
  RENDER.entrega = function () {
    var motivos = (S.config && S.config.motivos) || ['CICLO', 'ROTURA', 'CICLO INICIAL'];
    $('vista').innerHTML =
      '<h2 class="seccion-tit">Registrar entrega</h2>' +
      '<p class="seccion-sub">Cada entrega reinicia el reloj de esa prenda. No edita talles del padrón.</p>' +
      '<div class="card" style="max-width:640px">' +
        '<div class="grid2">' +
          '<label class="campo"><span>Legajo</span>' +
            '<input id="eLegajo" type="text" placeholder="N° de legajo"></label>' +
          '<label class="campo"><span>&nbsp;</span>' +
            '<button id="eBuscar" class="btn btn-ghost btn-block">Buscar operario</button></label>' +
        '</div>' +
        '<div id="eOperario"></div>' +
      '</div>';
    $('eBuscar').addEventListener('click', buscarParaEntrega);
    $('eLegajo').addEventListener('keydown', function (ev) { if (ev.key === 'Enter') buscarParaEntrega(); });

    function buscarParaEntrega() {
      var legajo = $('eLegajo').value.trim();
      if (!legajo) { toast('Escribí un legajo.', 'err'); return; }
      $('eOperario').innerHTML = cargando();
      api('datosOperario', { legajo: legajo }).then(function (op) {
        pintarFormEntrega(op, motivos);
      }).catch(function (e) { $('eOperario').innerHTML = '<div class="error">' + esc(e.message) + '</div>'; });
    }
  };

  function pintarFormEntrega(op, motivos) {
    var prendas = ['PANTALON', 'CAMISA', 'BOTIN'];
    var filas = prendas.map(function (pr) {
      var talle = (op.talles && op.talles[pr]) || '';
      var grupo = pr === 'BOTIN' ? (op.grupos && op.grupos.tipoBotin) : (op.grupos && op.grupos.color);
      return '<div class="item-prenda" data-prenda="' + pr + '">' +
        '<input type="checkbox" class="eInc" checked>' +
        '<input type="text" class="eTalle" value="' + esc(talle) + '" placeholder="talle">' +
        '<span class="muted">' + pr + (grupo ? ' · ' + esc(grupo) : '') + '</span>' +
      '</div>';
    }).join('');

    $('eOperario').innerHTML =
      '<div style="margin:6px 0 12px"><b>' + esc(op.nombre || '(sin nombre)') + '</b> · legajo ' + esc(op.legajo) +
        (op.activo ? '' : ' · <span class="badge sindato">INACTIVO</span>') + '</div>' +
      '<div class="grid2">' +
        '<label class="campo"><span>Fecha</span><input id="eFecha" type="date" value="' + hoyInput() + '"></label>' +
        '<label class="campo"><span>Motivo</span><select id="eMotivo">' +
          motivos.map(function (m) { return '<option value="' + esc(m) + '">' + esc(m) + '</option>'; }).join('') +
        '</select></label>' +
      '</div>' +
      '<label class="campo"><span>Prendas a entregar</span></label>' + filas +
      '<label class="campo"><span>Observaciones</span><textarea id="eObs" style="min-height:60px"></textarea></label>' +
      '<button id="eGuardar" class="btn btn-primary btn-block">Registrar entrega</button>';

    Array.prototype.forEach.call($('eOperario').querySelectorAll('.item-prenda'), function (row) {
      var chk = row.querySelector('.eInc');
      chk.addEventListener('change', function () { row.classList.toggle('off', !chk.checked); });
    });
    $('eGuardar').addEventListener('click', function () { guardarEntrega(op); });
  }

  function guardarEntrega(op) {
    var items = [];
    Array.prototype.forEach.call($('eOperario').querySelectorAll('.item-prenda'), function (row) {
      if (!row.querySelector('.eInc').checked) return;
      items.push({ prenda: row.getAttribute('data-prenda'), talle: row.querySelector('.eTalle').value.trim() });
    });
    if (!items.length) { toast('Elegí al menos una prenda.', 'err'); return; }
    var datos = {
      legajo: op.legajo,
      fecha: inputAFecha($('eFecha').value),
      motivo: $('eMotivo').value,

  function pintarCorrida(c, conAcciones) {
    var titulo = conAcciones ? 'Corrida actual' : 'Previsión';
    var rows = c.filas.map(function (f, i) {
      var clave = String(f.legajo).toUpperCase().replace(/\s+/g, '') + '|' + f.prenda;
      var excl = !!S.exclusiones[clave];
      var accCell = conAcciones
        ? '<td class="center"><input type="checkbox" class="exc" data-clave="' + esc(clave) + '" ' +
            (excl ? 'checked' : '') + (f.faltaTalle ? ' disabled' : '') + '></td>'
        : '';
      return '<tr class="' + (f.estado === 'CRITICO' ? 'row-critico ' : '') + (excl ? 'excluida' : '') + '">' +
        '<td>' + esc(f.legajo) + '</td>' +
        '<td>' + esc(f.nombre) + '</td>' +
        '<td>' + esc(f.centro || '') + '</td>' +
        '<td>' + esc(f.prenda) + '</td>' +
        '<td>' + esc(f.grupo || '') + '</td>' +
        '<td>' + (f.faltaTalle ? '<span class="badge sindato">falta</span>' : esc(f.talle)) + '</td>' +
        '<td class="num">' + (f.mesesUso === null ? '—' : f.mesesUso) + '</td>' +
        '<td>' + esc(f.ultimaEntrega || '—') + '</td>' +
        '<td>' + badgeEstado(f.estado, f.innegociable) + '</td>' + accCell +
      '</tr>';
    }).join('');

    var headers = ['Legajo', 'Nombre', 'Centro', 'Prenda', 'Grupo', 'Talle', 'Meses', 'Últ. entrega', 'Estado'];
    if (conAcciones) headers.push('No entregar');

    var acciones = conAcciones
      ? '<div class="fila-botones">' +
          '<button id="cPedido" class="btn btn-primary">Ver pedido →</button>' +
          '<button id="cLimpiar" class="btn btn-ghost">Limpiar exclusiones</button>' +
        '</div>'
      : '';

    $('vista').innerHTML =
      '<h2 class="seccion-tit">' + esc(titulo) + '</h2>' +
      '<p class="seccion-sub">' + esc(c.titulo || '') + '</p>' +
      chipsResumen(c.resumen) +
      (conAcciones ? '<p class="muted">Marcá “No entregar” las prendas que no van en esta compra. Las que faltan talle no se pueden pedir hasta cargarlo en Personal.</p>' : '') +
      (c.filas.length ? tabla(headers, rows) : vacio('No hay prendas en esta corrida.')) +
      acciones;

    if (conAcciones) {
      Array.prototype.forEach.call($('vista').querySelectorAll('.exc'), function (chk) {
        chk.addEventListener('change', function () {
          var k = chk.getAttribute('data-clave');
          if (chk.checked) S.exclusiones[k] = true; else delete S.exclusiones[k];
          chk.closest('tr').classList.toggle('excluida', chk.checked);
        });
      });
      $('cPedido').addEventListener('click', function () { ir('pedido'); });
      $('cLimpiar').addEventListener('click', function () { S.exclusiones = {}; ir('corrida'); });
    }
  }

  function pintarPedido(p) {
    var t = p.totales || {};

      '<div class="chip cobalto"><b>' + (t.unidades || 0) + '</b><span>Unidades</span></div>' +

      avisos += '<div class="card"><div class="card-tit">Quedaron afuera por falta de talle (' + p.sinTalle.length + ')</div>' +

      chips +
      (t.unidades ? matrices : vacio('No hay unidades para pedir (revisá exclusiones y talles).')) +
      proveedoresHtml(p.proveedores) +
      avisos +
      (t.unidades ? '<div class="fila-botones"><button id="pCongelar" class="btn btn-ok">Congelar esta compra</button>' +
        '<button id="pVolver" class="btn btn-ghost">← Volver a la corrida</button></div>' : '');

      $('pVolver').addEventListener('click', function () { ir('corrida'); });

  function matrizHtml(prenda, m) {
    if (!m || !m.total) return '';
    var th = ['<th>' + prenda + '</th>'].concat(m.grupos.map(function (g) { return '<th class="num">' + esc(g) + '</th>'; }))
      .concat(['<th class="num total-col">Total</th>']).join('');
    var body = m.talles.map(function (tl) {
      var celdas = m.grupos.map(function (g) {
        var n = (m.celdas[tl] && m.celdas[tl][g]) || 0;
        return '<td class="num celda">' + (n || '') + '</td>';
      }).join('');
      return '<tr><td><b>' + esc(tl) + '</b></td>' + celdas +
        '<td class="num total-col">' + (m.totalPorTalle[tl] || 0) + '</td></tr>';
    }).join('');
    var totalRow = '<tr class="total-row"><td>Total</td>' +
      m.grupos.map(function (g) { return '<td class="num">' + (m.totalPorGrupo[g] || 0) + '</td>'; }).join('') +
      '<td class="num total-col">' + m.total + '</td></tr>';
    return '<div class="card matriz"><div class="card-tit">' + prenda + ' — ' + m.total + ' u.</div>' +
      '<div class="tabla-wrap"><table class="tabla"><thead><tr>' + th + '</tr></thead><tbody>' +
      body + totalRow + '</tbody></table></div></div>';
  }

  function proveedoresHtml(pv) {
    if (!pv || !pv.hayPrecios) {
      return '<div class="card"><div class="card-tit">Proveedores</div>' +
        '<p class="muted">No hay precios cargados en la hoja PROVEEDORES. Cargalos para ver la comparación.</p></div>';
    }
    var rows = pv.proveedores.map(function (e) {
      var es = pv.ganador && pv.ganador.nombre === e.nombre;
      return '<tr>' +
        '<td>' + esc(e.nombre) + (es ? ' <span class="pill-ganador">ganador</span>' : '') + '</td>' +
        '<td class="num">' + (e.completo ? money(e.total) : '<span class="muted">incompleto</span>') + '</td>' +
        '<td class="muted">' + (e.completo ? '—' : 'no cotiza: ' + esc(e.faltan.join(', '))) + '</td>' +
      '</tr>';
    }).join('');

    var split = '';
    if (pv.split && pv.split.completo) {
      var det = Object.keys(pv.split.porPrenda).map(function (pr) {
        var x = pv.split.porPrenda[pr];
        return esc(pr) + ': ' + esc(x.proveedor) + ' (' + money(x.precio) + ')';
      }).join(' · ');
      split = '<p class="muted" style="margin-top:10px"><b>Split informativo</b> (cada prenda al más barato): ' +
        money(pv.split.total) + ' — ' + det +
        (pv.split.ahorro !== null ? ' · ahorro vs. ganador: ' + money(pv.split.ahorro) : '') +
        '.<br>No parte el pedido; es solo referencia.</p>';
    }

    return '<div class="card"><div class="card-tit">Comparación de proveedores</div>' +
      tabla(['Proveedor', 'Total (todo el pedido)', 'Detalle'], rows) + split + '</div>';
  }

  function aplicarCruce() {
    var cambios = [];
    Array.prototype.forEach.call($('rResultado').querySelectorAll('.cb-baja:checked'), function (chk) {
      cambios.push({ legajo: chk.getAttribute('data-legajo'), accion: 'DAR DE BAJA' });
    });
    Array.prototype.forEach.call($('rResultado').querySelectorAll('.cb-alta:checked'), function (chk) {
      cambios.push({
        legajo: chk.getAttribute('data-legajo'), accion: 'DAR DE ALTA',
        nombre: chk.getAttribute('data-nombre'), categoria: chk.getAttribute('data-categoria')
      });
    });
    if (!cambios.length) { toast('No seleccionaste ningún cambio.', 'err'); return; }
    if (!window.confirm('Vas a aplicar ' + cambios.length + ' cambio(s) en PERSONAL. ¿Confirmás?')) return;
    $('rAplicar').disabled = true;
    api('aplicarCambiosCruce', { cambios: cambios }).then(function (res) {
      var ok = res.resultados.filter(function (x) { return x.ok; }).length;
      toast('Aplicados ' + ok + '/' + res.resultados.length + ' cambios.', 'ok');
      $('rCruzar').click(); // re-cruzar para refrescar
    }).catch(function (e) { $('rAplicar').disabled = false; manejar(e); });
  }

        '<p class="seccion-sub">Roturas registradas y su antigüedad al romperse.</p>' + chips +
        (rep.eventos.length ? (
          '<div class="card"><div class="card-tit">Por operario</div>' +
            tabla(['Legajo', 'Nombre', 'Centro', 'Total', 'Tempr.', 'Tard.', 'P/C/B'], porLeg) + '</div>' +
          '<div class="card"><div class="card-tit">Por centro de costo</div>' +
            tabla(['Centro', 'Total', 'Tempr.', 'Tard.', 'P/C/B'], porCen) + '</div>' +
          '<div class="card"><div class="card-tit">Detalle de roturas</div>' +
            tabla(['Fecha', 'Legajo', 'Nombre', 'Prenda', 'Edad (m)', 'Tipo', 'Obs.'], eventos) + '</div>'
        ) : vacio('No hay roturas registradas (motivo ROTURA en entregas).'));
    }).catch(manejar);
  };

  /* ---------- Personal ---------- */
  RENDER.personal = function () {
    $('vista').innerHTML = '<h2 class="seccion-tit">Personal</h2>' +
      '<p class="seccion-sub">Padrón. Los talles se cargan acá (o en la hoja). La baja desactiva, no borra.</p>' +
      '<div class="fila-botones"><button id="pNuevo" class="btn btn-primary">+ Nuevo operario</button>' +
        '<label class="inline"><input type="checkbox" id="pSoloAct"> Ver solo activos</label></div>' +
      '<div id="pLista">' + cargando() + '</div>';
    $('pNuevo').addEventListener('click', function () { formOperario(null); });
    $('pSoloAct').addEventListener('change', cargarPersonal);
    cargarPersonal();
  };
  function cargarPersonal() {
    var soloActivos = $('pSoloAct') && $('pSoloAct').checked;
    api('listarPersonal', { soloActivos: soloActivos }).then(function (lista) {
      if (!lista.length) { $('pLista').innerHTML = vacio('No hay operarios.'); return; }
      var rows = lista.map(function (o) {
        var leg = o['LEGAJO'];
        var activo = /^s[ií]$|^true$|^x$|^1$/i.test(String(o['ACTIVO']).trim());
        return '<tr>' +
          '<td>' + esc(leg) + '</td>' +
          '<td>' + esc(o['APELLIDO Y NOMBRE']) + '</td>' +
          '<td>' + esc(o['CATEGORÍA'] || '') + '</td>' +
          '<td>' + esc(o['CENTRO DE COSTO'] || '') + '</td>' +
          '<td>' + esc(o['COLOR ROPA'] || '') + '</td>' +
          '<td class="muted">' + esc((o['TALLE PANT'] || '-') + '/' + (o['TALLE CAM'] || '-') + '/' + (o['TALLE BOTÍN'] || '-')) + '</td>' +
          '<td>' + (activo ? '<span class="badge elegible">activo</span>' : '<span class="badge sindato">inactivo</span>') + '</td>' +
          '<td><button class="btn btn-sm btn-ghost pEdit" data-legajo="' + esc(leg) + '">Editar</button> ' +
            '<button class="btn btn-sm pToggle" data-legajo="' + esc(leg) + '" data-activo="' + (activo ? '1' : '0') + '">' +
              (activo ? 'Baja' : 'Alta') + '</button></td>' +
        '</tr>';
      }).join('');
      $('pLista').innerHTML = tabla(['Legajo', 'Nombre', 'Categoría', 'Centro', 'Color', 'Talles P/C/B', 'Estado', ''], rows);

      // Guardar la lista cruda para edición rápida
      S._personal = {};
      lista.forEach(function (o) { S._personal[String(o['LEGAJO'])] = o; });

      Array.prototype.forEach.call($('pLista').querySelectorAll('.pEdit'), function (b) {
        b.addEventListener('click', function () { formOperario(S._personal[b.getAttribute('data-legajo')]); });
      });
      Array.prototype.forEach.call($('pLista').querySelectorAll('.pToggle'), function (b) {
        b.addEventListener('click', function () {
          var activar = b.getAttribute('data-activo') !== '1';
          api('setActivoOperario', { legajo: b.getAttribute('data-legajo'), activo: activar }).then(function () {
            toast(activar ? 'Reactivado.' : 'Dado de baja.', 'ok');
            cargarPersonal();
          }).catch(manejar);
        });
      });
    }).catch(function (e) { $('pLista').innerHTML = '<div class="error">' + esc(e.message) + '</div>'; });
  }

  function lst(key) { return (S.listas && S.listas[key]) || []; }
  function dataList(id, key) {
    return '<datalist id="' + id + '">' + lst(key).map(function (v) {
      return '<option value="' + esc(v) + '">';
    }).join('') + '</datalist>';
  }
  function campo(label, id, valor, listaId) {
    return '<label class="campo"><span>' + esc(label) + '</span>' +
      '<input id="' + id + '" type="text" value="' + esc(valor || '') + '"' +
      (listaId ? ' list="' + listaId + '"' : '') + '></label>';
  }

  function formOperario(o) {
    var esNuevo = !o;
    var v = function (k) { return o ? (o[k] || '') : ''; };
    var html =
      dataList('dlCat', 'CATEGORIA') + dataList('dlCen', 'CENTRO DE COSTO') +
      dataList('dlCol', 'COLOR ROPA') + dataList('dlBot', 'TIPO BOTIN') +
      dataList('dlTP', 'TALLE PANT') + dataList('dlTC', 'TALLE CAM') + dataList('dlTB', 'TALLE BOTIN') +
      '<div class="grid2">' +
        '<label class="campo"><span>Legajo</span><input id="oLegajo" type="text" value="' + esc(v('LEGAJO')) + '"' +
          (esNuevo ? '' : ' disabled') + '></label>' +
        campo('Apellido y nombre', 'oNombre', v('APELLIDO Y NOMBRE')) +
      '</div>' +
      '<div class="grid2">' +
        campo('Categoría', 'oCat', v('CATEGORÍA'), 'dlCat') +
        campo('Centro de costo', 'oCen', v('CENTRO DE COSTO'), 'dlCen') +
      '</div>' +
      '<div class="grid2">' +
        campo('Color ropa', 'oColor', v('COLOR ROPA'), 'dlCol') +
        campo('Tipo botín', 'oTipoBot', v('TIPO BOTÍN'), 'dlBot') +
      '</div>' +
      '<div class="grid2">' +
        campo('Talle pantalón', 'oTP', v('TALLE PANT'), 'dlTP') +
        campo('Talle camisa', 'oTC', v('TALLE CAM'), 'dlTC') +
      '</div>' +
      '<div class="grid2">' +
        campo('Talle botín', 'oTB', v('TALLE BOTÍN'), 'dlTB') +
        '<label class="campo"><span>&nbsp;</span><span class="muted">' +
          (esNuevo ? 'Se crea activo.' : 'El estado se cambia desde la lista.') + '</span></label>' +
      '</div>' +
      campo('Observaciones', 'oObs', v('OBSERVACIONES')) +
      '<div class="fila-botones"><button id="oGuardar" class="btn btn-primary">' +
        (esNuevo ? 'Crear operario' : 'Guardar cambios') + '</button>' +
        '<button id="oCancelar" class="btn btn-ghost">Cancelar</button></div>';

    abrirModal(esNuevo ? 'Nuevo operario' : 'Editar ' + v('LEGAJO'), html);
    $('oCancelar').addEventListener('click', cerrarModal);
    $('oGuardar').addEventListener('click', function () { guardarOperario(esNuevo); });
  }

  function guardarOperario(esNuevo) {
    var datos = {
      nombre: $('oNombre').value.trim(),
      categoria: $('oCat').value.trim(),
      centro: $('oCen').value.trim(),
      color: $('oColor').value.trim(),
      tipoBotin: $('oTipoBot').value.trim(),
      tallePant: $('oTP').value.trim(),
      talleCam: $('oTC').value.trim(),
      talleBotin: $('oTB').value.trim(),
      observaciones: $('oObs').value.trim()
    };
    $('oGuardar').disabled = true;
    var prom;
    if (esNuevo) {
      var legajo = $('oLegajo').value.trim();
      if (!legajo) { toast('El legajo es obligatorio.', 'err'); $('oGuardar').disabled = false; return; }
      datos.legajo = legajo;
      prom = api('crearOperario', { datos: datos });
    } else {
      prom = api('actualizarOperario', { legajo: $('oLegajo').value.trim(), datos: datos });
    }
    prom.then(function () {
      cerrarModal();
      toast(esNuevo ? 'Operario creado.' : 'Cambios guardados.', 'ok');
      cargarPersonal();
    }).catch(function (e) { $('oGuardar').disabled = false; manejar(e); });
  }
